Basic Usage #

Initializing #

html 
    <html>
    <head>
        <script src="https://github.com/chaosmail/dchart/raw/master/dist/dchart.min.js"></script>
    </head>
    <body>
        <div id="basicChart1"></div>
        <script>
            var baseChart1 = new dChart.LineChart({
                elem: 'basicChart1',
                ...
            });
        </script>
    </body>
    </html>

Chart Configuration #

html 
    <div id="basicChart2"></div>
javascript 
    var baseChart2 = new dChart.LineChart({
        elem: 'basicChart2',
        width:700,
        height:400,
        label: "Basic Chart: Chart Configuration",
        description: "The chart can be configured with some attributes",
        ...
    });

Axis Configuration #

    var baseChart3 = new dChart.LineChart({
        axis: {
            x: {
                label: "x Axis Label [m2]",
                align: "bottom",
                labelAlign: "right"
            },
            y: {
                label: "y Axis Label [s]",
                align: "left",
                labelAlign:"top"
            }
        },
        ...
    });
        
    var baseChart4 = new dChart.LineChart({
        axis: {
            marginTop: 40,
            marginRight: 40,
            x: {
                label: "x axis",
                align: "top",
                labelAlign: "left",
                gridStyle: {
                    stroke: "black",
                    strokeOpacity: 0.7
                }
            },
            y: {
                label: "y axis",
                align: "right",
                labelAlign:"center",
                gridStyle: {
                    stroke: "blueviolet",
                    strokeOpacity: 0.2
                },
                range:[0,1]
            }
        },
        ...
    });

Datasets #

Points #

    var pointsChart1 = new dChart.LineChart({
        ...
        dataSets: [
            {
                label: "Dataset: Points",
                data: [
                    {x:0,y:0},
                    {x:1,y:0.5},
                    {x:2,y:0.7},
                    {x:3,y:0.85},
                    {x:4,y:0.9},
                    {x:5,y:0.92}
                ]
            }
        ]
    });

Functions #

    var functionChart1 = new dChart.LineChart({
        ...
        dataSets: [
            {
                label: "Dataset: Function 'sin'",
                dataFn: {
                    fn: Math.sin,
                    min: 0,
                    max: 2*Math.PI,
                    step: Math.PI/360
                }
            },
            {
                label: "Dataset: Function 'cos'",
                dataFn: {
                    fn: Math.cos,
                    min: 0,
                    max: 2*Math.PI,
                    step: Math.PI/360
                }
            }
        ]
    });
        
    var functionChart2 = new dChart.LineChart({
        ...
        dataSets: [
            {
                label: "Dataset: Function 'random'",
                dataFn: {
                    fn: function() {
                        return Math.random();
                    },
                    min: 0,
                    max: 30,
                    step: 1
                }
            }
        ]
    });

Chart Types #

Line Charts #

    var lineChart1 = new dChart.LineChart({
        elem: 'lineChart1',
        width: 700,
        height: 400,
        label: "My awesome Chart",
        description: "This awesome chart lib is using D3.js",
        axis: {
            x: {
                label: "x-Axis",
                align: "bottom",
                labelAlign: "right"
            },
            y: {
                label: "y-Axis",
                align: "left",
                scale: "sqrt",
                labelAlign: "top",
                gridStyle: {
                    stroke: "black",
                    strokeOpacity: 0.25
                }
            }
        },
        dataSets:[
            {
                label: "Dataset 1",
                lineStyle: {
                    stroke: "mediumslateblue",
                    strokeWidth: 2,
                    strokeDasharray: "5,5"
                },
                dotStyle: {
                    stroke: "mediumslateblue",
                    strokeWidth: 2,
                    fill: "#fff"
                },
                interpolate: "linear",
                data: [
                    {x:0,y:0},
                    {x:1,y:0.5},
                    {x:2,y:0.7},
                    {x:3,y:0.85},
                    {x:4,y:0.9},
                    {x:5,y:0.92}
                ]
            },
            {
                label: "Dataset 2: Function",
                lineStyle: {
                    stroke: "red",
                    strokeWidth: 2
                },
                interpolate:'cardinal',
                dataFn: {
                    fn: function() { return Math.random() },
                    min: 0,
                    max: 5,
                    step: 0.25
                }
            }
        ]
    });
